镜像操作

```
# 查询镜像
docker search xxx

# 拉取镜像 latest(获取最新版本)
docker pull xxx:latest

# 列出本地镜像
docker images
```

容器操作

```
# 创建并启动一个容器
# 以nginx为例 容器名称为nginx 
# 8080 本机端口
# 80 容器端口
# nginx 依赖的镜像
docker run --name nginx -p 8080:80 -d nginx

# 列出正在运行的容器
# -a :显示所有的容器，包括未运行的。
# -f :根据条件过滤显示的内容。
# --format :指定返回值的模板文件。
# -l :显示最近创建的容器。
# -n :列出最近创建的n个容器。
# --no-trunc :不截断输出。
# -q :静默模式，只显示容器编号。
# -s :显示总的文件大小。
docker ps
docker ps -a 
docker ps -n 5
docker ps -a -q
docker ps -a -s

# 启动容器 
docker start [容器名称] / [容器ID]

# 停止容器
docker stop [容器名称] / [容器ID]

# 移除容器 (需先停止容器)
docker rm [容器名称] / [容器ID]

# 查看网络端口 (以上面的nginx容器为例)
docker port nginx 80

# 查看容器日志 (-f 类似 tail -f 实时输出 crtl+C 退出 )
docker logs -f [容器名称] / [容器ID]

# 查看容器进程
docker top [容器名称] / [容器ID]

# 查了容器配置和状态信息
docker inspect [容器名称] / [容器ID]


```
